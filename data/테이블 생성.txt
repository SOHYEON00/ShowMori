CREATE TABLE USER_T (
ID varchar(10) NOT NULL,
PW CHAR(10) NOT NULL,
U_NAME VARCHAR(10) NOT NULL,
U_PHONE VARCHAR(20) NOT NULL,

PRIMARY KEY (ID)
);

CREATE TABLE POST_T(
S_PRM INT NOT NULL AUTO_INCREMENT,
ID varchar(10) NOT NULL,
S_POSTER VARCHAR(255) NOT NULL, 
S_SYNOP VARCHAR(255) NOT NULL,
S_TITLE VARCHAR(255) NOT NULL,
S_GOALSUM INT DEFAULT 0,
S_DEADLINE DATE NOT NULL,
START_DAY DATE NOT NULL,
LAST_DAY DATE NOT NULL,

PRIMARY KEY(S_PRM),
FOREIGN KEY (ID) REFERENCES USER_T(ID)
);


CREATE TABLE S_DATE_T(
S_DATE_PRM INT NOT NULL AUTO_INCREMENT,
S_PRM INT NOT NULL,
DAY DATE,

PRIMARY KEY (S_DATE_PRM),
FOREIGN KEY (S_PRM) REFERENCES POST_T(S_PRM)
);


CREATE TABLE REWARD_T(
R_PRM INT NOT NULL AUTO_INCREMENT ,
D_MONEY INT NOT NULL,
D_REWARD VARCHAR(40),

PRIMARY KEY (R_PRM)
);

CREATE TABLE D_INFO_T(
D_PRM INT NOT NULL AUTO_INCREMENT,
S_PRM INT NOT NULL,
ID varchar(10) NOT NULL,
D_MONEY INT,
D_DATE DATE,

PRIMARY KEY (D_PRM),
FOREIGN KEY (S_PRM) REFERENCES POST_T (S_PRM),
FOREIGN KEY (ID) REFERENCES USER_T (ID)
);
